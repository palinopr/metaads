# Error Handling Integration Example\n\nHere's how to integrate the comprehensive error handling system into your Meta Ads Dashboard:\n\n## 1. Initialize in Layout\n\n```typescript\n// app/layout.tsx\nimport { GlobalErrorHandler } from '@/components/global-error-handler'\nimport { initializeErrorHandling } from '@/lib/error-handling'\n\nexport default function RootLayout({ children }) {\n  useEffect(() => {\n    // Initialize error handling system\n    initializeErrorHandling({\n      enableDebugMode: process.env.NODE_ENV === 'development',\n      enableNotifications: true,\n      enableOfflineQueue: true,\n      debugSettings: {\n        verboseLogging: true,\n        captureNetworkLogs: true,\n        capturePerformanceMetrics: true,\n        trackUserActions: true\n      }\n    })\n  }, [])\n\n  return (\n    <html>\n      <body>\n        <GlobalErrorHandler />\n        <ThemeProvider>\n          {children}\n        </ThemeProvider>\n      </body>\n    </html>\n  )\n}\n```\n\n## 2. Enhance Meta API Calls\n\n```typescript\n// lib/meta-api-enhanced.ts\nimport { ErrorRecovery, RecoveryStrategies, ApplicationError, ErrorCategory, ErrorSeverity } from '@/lib/error-handling'\n\nexport class MetaAPIEnhanced {\n  async fetchCampaigns(accountId: string) {\n    return await RecoveryStrategies.handleAPIError(\n      async () => {\n        const response = await fetch(`/api/meta/campaigns?account_id=${accountId}`)\n        \n        if (!response.ok) {\n          throw new ApplicationError(\n            `Failed to fetch campaigns: ${response.statusText}`,\n            {\n              category: ErrorCategory.API,\n              severity: response.status >= 500 ? ErrorSeverity.HIGH : ErrorSeverity.MEDIUM,\n              code: `META_API_${response.status}`,\n              context: {\n                endpoint: '/api/meta/campaigns',\n                accountId,\n                statusCode: response.status\n              }\n            }\n          )\n        }\n        \n        return response.json()\n      },\n      {\n        maxRetries: 3,\n        retryCondition: (error, attempt) => {\n          // Retry on server errors and rate limits\n          return error instanceof ApplicationError &&\n                 (error.context?.statusCode >= 500 || error.context?.statusCode === 429)\n        }\n      }\n    )\n  }\n\n  async refreshToken() {\n    return await RecoveryStrategies.handleNetworkError(\n      async () => {\n        // Token refresh logic\n        const response = await fetch('/api/auth/refresh')\n        if (!response.ok) {\n          throw new ApplicationError(\n            'Token refresh failed',\n            {\n              category: ErrorCategory.AUTHENTICATION,\n              severity: ErrorSeverity.HIGH\n            }\n          )\n        }\n        return response.json()\n      }\n    )\n  }\n}\n```\n\n## 3. Wrap Components with Error Boundaries\n\n```typescript\n// app/dashboard/page.tsx\nimport { EnhancedErrorBoundary } from '@/lib/error-handling'\n\nexport default function DashboardPage() {\n  return (\n    <EnhancedErrorBoundary\n      level=\"page\"\n      componentName=\"DashboardPage\"\n      enableRetry={true}\n      maxRetries={3}\n    >\n      <DashboardContent />\n    </EnhancedErrorBoundary>\n  )\n}\n\n// components/campaign-card.tsx\nimport { withErrorBoundary, useErrorBoundary } from '@/lib/error-handling'\n\nfunction CampaignCard({ campaign }) {\n  const { captureError } = useErrorBoundary()\n  \n  const handleUpdate = async () => {\n    try {\n      await updateCampaign(campaign.id)\n    } catch (error) {\n      captureError(error)\n    }\n  }\n  \n  return (\n    <Card>\n      <CardHeader>\n        <CardTitle>{campaign.name}</CardTitle>\n      </CardHeader>\n      <CardContent>\n        <Button onClick={handleUpdate}>Update</Button>\n      </CardContent>\n    </Card>\n  )\n}\n\n// Wrap with error boundary\nexport default withErrorBoundary(CampaignCard, {\n  level: 'component',\n  enableRetry: true,\n  maxRetries: 2,\n  componentName: 'CampaignCard'\n})\n```\n\n## 4. Add Error Analytics to Admin Panel\n\n```typescript\n// app/admin/errors/page.tsx\nimport { ErrorAnalyticsDashboard } from '@/lib/error-handling'\n\nexport default function ErrorAnalyticsPage() {\n  return (\n    <div className=\"container mx-auto py-6\">\n      <h1 className=\"text-3xl font-bold mb-6\">Error Analytics</h1>\n      <ErrorAnalyticsDashboard />\n    </div>\n  )\n}\n```\n\n## 5. Custom Error Handling for Forms\n\n```typescript\n// components/campaign-form.tsx\nimport { ErrorUtils, ValidationError } from '@/lib/error-handling'\n\nexport function CampaignForm() {\n  const handleSubmit = ErrorUtils.makeSafe(\n    async (formData) => {\n      // Validate form data\n      if (!formData.name) {\n        throw new ValidationError(\n          'Campaign name is required',\n          'name',\n          formData.name\n        )\n      }\n      \n      // Submit form\n      await submitCampaign(formData)\n    },\n    undefined, // no fallback\n    { component: 'CampaignForm', action: 'submit' }\n  )\n  \n  return (\n    <form onSubmit={handleSubmit}>\n      {/* form fields */}\n    </form>\n  )\n}\n```\n\n## 6. Custom Notification Rules\n\n```typescript\n// app/admin/settings/error-notifications.tsx\nimport { notificationSystem } from '@/lib/error-handling'\n\nexport function setupCustomNotifications() {\n  // Meta API specific alerts\n  notificationSystem.addRule({\n    id: 'meta-token-expiry',\n    name: 'Meta Token Expiry Alert',\n    threshold: {\n      severity: ErrorSeverity.HIGH,\n      category: ErrorCategory.AUTHENTICATION,\n      count: 1,\n      timeWindow: 1,\n      cooldown: 60\n    },\n    channels: [\n      { type: 'toast', enabled: true, config: {} },\n      { type: 'webhook', enabled: true, config: { url: process.env.SLACK_WEBHOOK_URL } }\n    ],\n    enabled: true\n  })\n  \n  // Campaign load failures\n  notificationSystem.addRule({\n    id: 'campaign-load-failures',\n    name: 'Campaign Load Failure Spike',\n    threshold: {\n      severity: ErrorSeverity.MEDIUM,\n      category: ErrorCategory.API,\n      count: 10,\n      timeWindow: 5,\n      cooldown: 15\n    },\n    channels: [\n      { type: 'toast', enabled: true, config: {} }\n    ],\n    enabled: true\n  })\n}\n```\n\nThis integration provides:\n\n✅ **Zero unhandled errors** - All errors are caught and tracked\n✅ **Comprehensive monitoring** - Real-time error analytics and health checks  \n✅ **Automatic recovery** - Retry logic and fallback strategies\n✅ **Offline support** - Error queuing when connection is lost\n✅ **Developer tools** - Debug mode with detailed logging\n✅ **User notifications** - Smart alerting based on error patterns\n✅ **Performance monitoring** - Track error impact on user experience\n\nThe system is now ready to handle all error scenarios in your Meta Ads Dashboard!\n"
